int maxArea(int* height, int heightSize) {
    int left = 0;              // 1️⃣ 左指针
    int right = heightSize - 1; // 2️⃣ 右指针
    int maxArea = 0;           // 3️⃣ 最大水量
    
    while (left < right) {     // 4️⃣ 双指针向中间移动
        // 5️⃣ 计算当前水量
        int width = right - left;
        int h = (height[left] < height[right]) ? height[left] : height[right];
        int area = width * h;
        
        // 6️⃣ 更新最大值
        if (area > maxArea) {
            maxArea = area;
        }
        
        // 7️⃣ 移动短板（核心！）
        if (height[left] < height[right]) {
            left++;   // 左板更短，右移左指针
        } else {
            right--;  // 右板更短或相等，左移右指针
        }
    }
    
    return maxArea;  // 8️⃣ 返回最大水量
}
