## 代码

```c
#include <stdbool.h>  // 1️⃣ 引入布尔类型（true/false）
#include <string.h>   // 2️⃣ 引入字符串函数（比如strlen）

bool isAlienSorted(char** words, int wordsSize, char* order) {
    // 3️⃣ 定义函数：判断外星语单词是否按顺序排列
    // 参数说明：
    // words：单词数组（比如["hello","leetcode"]）
    // wordsSize：单词个数（比如2）
    // order：外星语字母顺序（比如"hlabc..."）

    int rank[26];  // 4️⃣ 创建一个数组，存26个字母的“权重分”
    // 为什么是26？因为外星语只用26个小写字母

    // 5️⃣ 给每个字母打分：外星语中第i个字母的分数是i
    for (int i = 0; i < 26; i++) {
        // order[i]是当前外星语字母（比如order[0]是'h'）
        // order[i] - 'a'：把字母转换成0-25的数字（'a'→0，'b'→1…）
        // 比如order[0]是'h'，'h'-'a'=7？不对，等一下——哦不，这里rank[order[i]-'a']=i的意思是：
        // 外星语中第i个字母（order[i]）的权重是i。比如order[0]是'h'，那么rank['h'-'a']=0，即'h'的分数是0。
        rank[order[i] - 'a'] = i;
    }

    // 6️⃣ 两两比较相邻单词（比如第1个和第2个，第2个和第3个…）
    for (int i = 0; i < wordsSize - 1; i++) {
        char* word1 = words[i];       // 当前单词（比如"hello"）
        char* word2 = words[i + 1];   // 下一个单词（比如"leetcode"）
        int len1 = strlen(word1);      // 当前单词长度（"hello"是5）
        int len2 = strlen(word2);      // 下一个单词长度（"leetcode"是8）
        bool isSame = true;            // 标记两个单词是否完全相同（默认是）

        // 7️⃣ 逐字符比较两个单词（直到其中一个结束）
        for (int j = 0; j < len1 && j < len2; j++) {
            char c1 = word1[j];  // 当前单词的第j个字符（比如"hello"[0]是'h'）
            char c2 = word2[j];  // 下一个单词的第j个字符（比如"leetcode"[0]是'l'）

            if (c1 != c2) {  // 8️⃣ 找到第一个不同的字符
                // 比较它们的权重分：如果c1的分数>c2的分数，说明顺序错了
                if (rank[c1 - 'a'] > rank[c2 - 'a']) {
                    return false;  // 直接返回false，不用继续比较
                }
                isSame = false;  // 标记为“不是完全相同的单词”
                break;            // 跳出当前循环（后面的字符不用比了）
            }
        }

        // 9️⃣ 处理前缀情况：如果两个单词完全相同，但第一个更长（比如"apple"和"app"）
        if (isSame && len1 > len2) {
            return false;  // 短的应该排在前面，所以顺序错了
        }
    }

    // 🔟 所有比较都通过，返回true
    return true;
}
```

## 核心思路（通俗版）

1. **给外星语字母打分** ：比如外星语中 `h`是第1个字母，给它打0分；`l`是第2个，打1分……这样每个字母都有一个“权重分”。
2. **按分数比较单词** ：就像英语字典按字母顺序排，外星语按“权重分”排。比如比较两个单词时，第一个不同字母的分数小的单词排在前面。
3. **处理特殊情况** ：如果一个单词是另一个的前缀（比如 `"app"`和 `"apple"`），短的应该排在前面。
